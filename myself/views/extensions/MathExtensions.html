interface Math {
	/**
	 * 得到[min, max)之间的浮点随机数
	 */
	randomFloat(min: number, max: number): number;
	/**
	 * 得到[min, max]之间的整数随机数，注意max值可以取到
	 */
	randomInteger(min: number, max: number): number;
	/**
	 * 限制value的值在[min,max]之间
	 */
	clamp(value: number, min: number, max: number): number;
}
